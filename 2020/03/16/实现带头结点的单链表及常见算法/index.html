#include "LinkList.h"
#include <stdio.h>

int main()
{
    Node Link;
    InitList(&Link);
    for (int i = 0; i < 10; i++)
    {
        ListInsertHead(&Link,i);
    }
    Show(&Link);
    printf("%d\n", ListLength(&Link));
    printf("%d\n", GetElem(&Link,3));
    printf("%d\n",LocateElem(&Link,5));
    Node * _5PosPrior = GetPosPrior(&Link, 5);
    printf("pos 5 prior node data : %d\n",(*_5PosPrior).data);
    ListInsert(&Link,4,999);
    Show(&Link);
    ListInsertTail(&Link,666);
    Show(&Link);
    ListDeletePos(&Link,4);
    Show(&Link);
    ListDeleteHead(&Link);
    Show(&Link);
    ListDeleteTail(&Link);
    Show(&Link);
    ListInsert(&Link, 5,1024);
    Show(&Link);
    ListDeleteData(&Link,1024);
    Show(&Link);
    Node* kToLast = Find_k_to_last(&Link,5);
    printf("linklist %d to last is %d\n",5,kToLast->data);
    Node * tmp =GetPosPrior(&Link,2);//获得pos位置节点的前驱节点
    DeleteP(&Link, tmp);
    Show(&Link);
    reverse(&Link);
    Show(&Link);
    return 0;
}

#pragma 

/*------------------------------------------------------------
// 链表结构的定义
------------------------------------------------------------*/

typedef int ElemType;


typedef struct _Node
{
    ElemType data;                // 元素数据
    struct _Node* next;            // 链表中结点元素的指针
}Node, *LinkList;

/*------------------------------------------------------------
// 链表的基本操作
------------------------------------------------------------*/

void InitList(LinkList head); //初始化
void DestroyList(LinkList head);  //销毁
bool IsListEmpty(LinkList head);//判空
int  ListLength(LinkList head);//求链表长度
ElemType GetElem(LinkList head, int pos); //得到链表的第pos个元素
int LocateElem(LinkList head, ElemType data);  //得到链表指定元素的位置
Node* GetPosPrior(LinkList head, int pos);//获得pos位置节点的前驱节点
void Show(LinkList head);  //打印单链表
void ListInsert(LinkList head, int pos, ElemType data);//在指定位置插入元素
void ListInsertHead(LinkList head,ElemType data);  //在链表的头部插入元素
void ListInsertTail(LinkList head, ElemType data); //在链表的尾部插入元素
void ListDeletePos(LinkList head, int pos);//删除指定位置的元素
void ListDeleteHead(LinkList head);//删除单链表的第一个元素
void ListDeleteTail(LinkList head);//删除单链表的最后一个元素
void ListDeleteData(LinkList head, ElemType data);  ////删除与data相等数据的元素
/*面经题
1、找倒数第k个结点O(n)
2、O(1)删除结点p（不是尾结点）
3、单链表的原地逆置
4、判断两个单链表是否相交，找到相交的一个结点
5、判断单链表是否有环，找到入环的第一个结点
*/
Node* Find_k_to_last(LinkList head, int k);  //查找到链表中的倒数第K个节点O(n)
void DeleteP(LinkList head, Node* pfind);   //O(1)删除结点p（不是尾结点）
LinkList reverse(LinkList head); //单链表的原地逆置
Node* IsIntersect(LinkList head1, LinkList head2);//判断两个单链表是否相交，找到相交的一个结点
Node* IsLoop(LinkList head);  //判断单链表是否有环,找到入环的第一个结点

#include "LinkList.h"
#include <stdio.h>
#include <stdlib.h>
#include <malloc.h>
#include <memory.h>
#include <assert.h>


/*------------------------------------------------------------
操作目的：    初始化链表
初始条件：    无
操作结果：    构造一个空的线性表
函数参数：
        LinkList head    待初始化的线性表
返回值：
        void    
------------------------------------------------------------*/
void InitList(LinkList head)
{
    if(head == NULL)
        exit(-1);
    head->next = NULL;
    head->data = 0;
}

/*------------------------------------------------------------
操作目的：    销毁链表
初始条件：    线性表head已存在
操作结果：    线性表head不存在
函数参数：
        LinkList head    待销毁的线性表
返回值：
        无
------------------------------------------------------------*/
void DestroyList(LinkList head)
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    while (head->next)
    ListDeleteHead(head);
}

/*------------------------------------------------------------
操作目的：    判断链表是否为空
初始条件：    线性表head已存在
操作结果：    若head为空表，则返回true，否则返回false
函数参数：
        LinkList head    待判断的线性表
返回值：
        bool        是否为空
------------------------------------------------------------*/
bool IsListEmpty(LinkList head)
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    return head->next == NULL ? true : false;

}

/*------------------------------------------------------------
操作目的：    得到链表的长度
初始条件：    线性表head已存在
操作结果：    返回L中数据元素的个数
函数参数：
        LinkList head    线性表head
返回值：
        int            数据元素的个数
------------------------------------------------------------*/
int  ListLength(LinkList head)
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    return head->data;
}




/*------------------------------------------------------------
操作目的：    得到链表的pos位置元素
初始条件：    线性表L已存在，0<=i<=ListLength(head)
操作结果：    返回找打的元素
函数参数：
        LinkList head    线性表head
        int i        数据元素的位置
返回值：
        ElemType 返回元素数据
------------------------------------------------------------*/
ElemType GetElem(LinkList head, int pos)
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    Node* tmp = head->next;
    while (pos > 0)
    {
        tmp = tmp->next;
        pos--;
    }
    return tmp->data;
}

/*------------------------------------------------------------
操作目的：    得到链表指定元素的位置
初始条件：    线性表head已存在
操作结果：    返回查找到的和指定data相等的元素的下标
            若这样的元素不存在则返回0。
函数参数：
        LinkList head    线性表head
        ElemType data    查找的元素
返回值：
        int   返回元素的位置
------------------------------------------------------------*/
int LocateElem(LinkList head, ElemType data)
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    Node* tmp = head->next;
    unsigned count = 0;
    while (tmp)
    {
        if(data == tmp->data)
            return count;
        else
        {
            count++;
            tmp = tmp->next;
        }    
    }
    return -1;
}

/*------------------------------------------------------------
操作目的：    返回链表中pos位置的前驱节点
初始条件：    线性表head已存在
操作结果：    返回链表中pos位置的前驱节点的地址
函数参数：
        LinkList head        线性表head
int pos           链表位置
返回值：
        返回前驱节点
------------------------------------------------------------*/
Node* GetPosPrior(LinkList head, int pos)
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    if (pos<0 || pos>head->data)
    {
        printf("pos error");
        exit(0);
    }
    Node* tmp = head;
    while (pos > 0)
    {
        tmp = tmp->next;
        pos--;
    }
    return tmp;
}
/*------------------------------------------------------------
操作目的：    打印链表
初始条件：    线性表head已存在
操作结果：    依次打印链表中所有元素的数据域
函数参数：
        LinkList head        线性表head
返回值：
        void
------------------------------------------------------------*/
void Show(LinkList head)
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    Node* tmp = head->next;
    while (tmp)
    {
        printf("%d\t", tmp->data);
        tmp = tmp->next;
    }
    printf("\n");

}


/*------------------------------------------------------------
操作目的：    在链表的指定位置插入结点，插入位置i表示在第i个
            元素之前插入
初始条件：    线性表head已存在，0<=i<=ListLength(head)
操作结果：    在head中第pos个位置插入新的数据元素data，head的长度加1
函数参数：
        LinkList head    线性表head
        int pos        插入位置
        ElemType data    待插入的数据元素
返回值：
        void
------------------------------------------------------------*/
void ListInsert(LinkList head, int pos, ElemType data)
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    if (pos<0 || pos>head->data)
    {
        printf("pos error");
        exit(0);
    }
    Node* tmp = (Node *)malloc(sizeof(Node));
    if (tmp == NULL)
    {
        printf("apply fail");
        exit(-1);
    }
    tmp->data = data;
    Node * PosPrior = GetPosPrior(head,pos);
    tmp->next = PosPrior->next;
    PosPrior->next = tmp;
    head->data++;
}


/*------------------------------------------------------------
操作目的：    在链表的头部插入元素
初始条件：    线性表L已存在
操作结果：    在链表的头部插入元素，head的长度加1
函数参数：
        LinkList head    线性表head
        ElemType data    待插入的数据元素
返回值：
        bool        操作是否成功
------------------------------------------------------------*/
void ListInsertHead(LinkList head, ElemType data)
{

    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    ListInsert(head, 0, data);
}


/*------------------------------------------------------------
操作目的：    在链表的尾部位置插入节点
初始条件：    线性表head已存在
操作结果：    在L中尾部插入节点，head的长度加1
函数参数：
        LinkList head    线性表head
        ElemType data    待插入的数据元素
返回值：
        无
------------------------------------------------------------*/
void ListInsertTail(LinkList head, ElemType data)
{

    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    ListInsert(head, head->data, data);
}




/*------------------------------------------------------------
操作目的：    删除链表的第i个结点
初始条件：    链表存在
操作结果：    删除head的第i个数据元素
函数参数：
        LinkList head    线性表head
        int pos       删除位置

返回值：
        无返回值
------------------------------------------------------------*/
void ListDeletePos(LinkList head, int pos)
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    if (pos<0 || pos>head->data)
        exit(0);
    Node * tmp = GetPosPrior(head,pos);
    Node* PosNode = tmp->next;
    tmp->next = PosNode->next;
    free(PosNode);
    head->data--;
}

/*------------------------------------------------------------
操作目的：    删除链表的头节点
初始条件：    线性表head已存在且非空
操作结果：    删除head的头节点
函数参数：
        LinkList head    线性表head
返回值：
        无返回值
------------------------------------------------------------*/
void ListDeleteHead(LinkList head)
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    ListDeletePos(head, 0);
}


/*------------------------------------------------------------
操作目的：    删除链表的尾节点
初始条件：    线性表head已存在且非空
操作结果：    删除head的尾节点
函数参数：
        LinkList head    线性表head
返回值：
        无返回值
------------------------------------------------------------*/
void ListDeleteTail(LinkList head)
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    ListDeletePos(head, head->data - 1);

}


/*------------------------------------------------------------
操作目的：    删除链表数据域为data的节点
初始条件：    线性表head已存在且非空
操作结果：    删除链表数据域为data的节点
函数参数：
        LinkList head    线性表head
        ElemType data    被删除的数据元素值
返回值：
        无返回值
------------------------------------------------------------*/
void ListDeleteData(LinkList head, ElemType data)   //删除与data相等数据的元素
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    Node* tmp = head->next;
    int count = 0;
    while (tmp)
    {
        if (tmp->data == data)
        {
            Node* tmp = GetPosPrior(head,count);
            Node* PosNode = tmp->next;
            tmp->next = PosNode->next;
            free(PosNode);
            return;
        }
        else
        {
            tmp = tmp->next;
            count++;
        }    
    }
}

/*------------------------------------------------------------
操作目的：    查找链表中倒数第k和节点
初始条件：    线性表head已存在且非空
操作结果：    查找链表中倒数第k和节点
函数参数：
        LinkList head    线性表head
        int  k  倒数第k个元素
返回值：
        无返回值
------------------------------------------------------------*/
Node* Find_k_to_last(LinkList head, int k)
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    if (k < 1)
    {
        printf("pos error");
        exit(0);
    }
    Node* fast = head->next;
    Node* slow = head->next;
    while (k > 0 && fast)
    {
        fast = fast->next;
        k--;
    }

    while (fast)
    {
        fast = fast->next;
        slow = slow->next;
    }
    return slow;

}

void DeleteP(LinkList head,Node * pfind)
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    if (pfind->next == NULL)
    {
        while (head->next != pfind)
            head = head->next;
        head->next = pfind->next;
        free(pfind);
        pfind = NULL;
    }
    else
    {
        Node* p = pfind->next;
        pfind->data = pfind->next->data;
        pfind->next = pfind->next->next;
        free(p);
        pfind = NULL;
    }
}

LinkList reverse(LinkList head) //单链表的原地逆置
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    Node* p;
    Node* s;
    p = head->next;
    s = p->next;
    head->next = NULL;
    while (p)
    {
        p->next = head->next;
        head->next = p;
        p = s;
        if (p != NULL)
            s = p->next;
    }
    return head;
}

Node* IsIntersect(LinkList head1, LinkList head2)//判断两个单链表是否相交，找到相交的一个结点
{
    if (head1 == NULL || head2 == NULL)  return NULL;
    int distance = 0;
    Node* p = head1->next;
    Node* q = head2->next;
    distance = head1->data > head2->data ? head1->data - head2->data : head2->data - head1->data;
    if (head1->data > head2->data)
    {
        while (distance)
        {
            p = p->next;
            distance--;
        }
        while (p)
        {
            if (p == q)  return p;
            p = p->next;
            q = q->next;
        }
    }
    if(head1->data <= head2->data)
    {
        while (distance)
        {
            q = q->next;
            distance--;
        }
        while (q)
        {
            if (p == q)  return p;
            p = p->next;
            q = q->next;
        }
    }
    return NULL;
}

Node* IsLoop(LinkList head)//判断单链表是否有环，找到入环的第一个结点
{
    assert(head != NULL);
    if (head == NULL)
    {
        printf("%s %d error", __FUNCTION__, __LINE__);
        exit(0);
    }
    Node* p = head;
    Node* q = head;
    while (1)
    {
        p = p->next;
        if (q == NULL || q->next == NULL)
            break;
        q = q->next->next;
        if (p == q)
            break;
    }
    if (p != q)
        return NULL;
    else
        p = head->next;
    while (p != q)
    {
        p = p->next;
        q = q->next;
    }
    return p;
}
