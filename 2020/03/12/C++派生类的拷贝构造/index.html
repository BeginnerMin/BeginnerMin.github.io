派生类的拷贝构造

拷贝构造，也是一种构造函数（不要拿拷贝构造不当构造，是己有对象构造新对象），也没有被继承下来。

引例

父类和子类的拷贝构造都是默认的情况

#include <iostream>
using namespace std;


class A
{
public:
    A()
    {
        cout << "A()" << endl;
    }

};

class B : public A
{
public:
    B()
    {
        cout << "B()" << endl;
    }

};


int main()
{
    B b;
    B bb(b);
    return 0;
}

运行结果为：

父类和子类的拷贝构造都是默认的情况下子类就会先调用父类的默认拷贝构造，然后调用子类的拷贝构造

父类和子类的拷贝构造都自实现的情况

#include <iostream>
using namespace std;


class A
{
public:
    A()
    {
        cout << "A()" << endl;
    }

    A(const A& another)
    {
        cout << "A(const A & another)" << endl;
    }
};

class B: public A
{
public:
    B()
    {
        cout << "B()" << endl;
    }

    B(const B& another)
    {
        cout << "B(const B & another)" << endl;
    }
};


int main()
{
    B b;
    B bb(b);
    return 0;
}

运行结果为：

当两个类中都自实现拷贝构造的时候，
在类A和类B完成构造之后，会继续调用A的构造，然后调用B的拷贝构造，整个过程没有调用A的拷贝构造。

那么为什么不涉及A的拷贝构造呢？ 我们在下面进行说明。

在父类自实现拷贝构造，子类使用默认的拷贝构造

#include <iostream>
using namespace std;


class A
{
public:
    A()
    {
        cout << "A()" << endl;
    }

    A(const A& another)
    {
        cout << "A(const A & another)" << endl;
    }
};

class B: public A
{
public:
    B()
    {
        cout << "B()" << endl;
    }


};


int main()
{
    B b;
    B bb(b);
    return 0;
}

运行结果为：

在父类自实现拷贝构造，子类使用默认的拷贝构造的情况下，子类没有自实现拷贝构造的时候，会调用父类的拷贝构造，而不会去调用父类的构造器，然后默认调用子类的拷贝构造。

我们的目的就是要去调用父类的拷贝构造。

在父类使用默认的拷贝构造，子类使用自实现的拷贝构造

#include <iostream>

引例总结

当子类不自实现拷贝构造的时候会默认调用父类的拷贝构造。

如果子类自实现拷贝构造不作特殊处理此时只会调用父类的构造器。不会调用父类的拷贝构造器，这种情况就失去了拷贝的意义。

直观的说明

#include <iostream>
using namespace std;


class A
{
public:
    A(int x = 22)
    {
        a = x;
        cout << "A()" << endl;
    }

    A(const A& another)
    {
        this->a = another.a;
        cout << "A(const A & another)" << endl;
    }
    int a;
};

class B : public A
{
public:
    B(int z , int y)
        :A(z)
    {
        b = y;
        cout << "B()" << endl;
    }

    B(const B& another)
    {
        this->b = another.b;
        cout << "B(const B & another)" << endl;
    }
    int b;
};


int main()
{
    B b(666, 100);
    B bb(b);
    cout << bb.a << endl;
    cout << bb.b << endl;
    return 0;
}

运行结果为：

我们可以看到，我们给b对象传递的参数是666和100但是经过拷贝构造之后到bb对象的时候就直接变成了22和100，我们通过代码进行分析就可以看出来，在进行完父类A和子类B的构造之后，子类B调用父类A的构造器，并没有调用父类A的拷贝构造，然后执行B类的拷贝构造，所以最终的结果并没有实现父类的拷贝构造。那么这个时候的拷贝构造就是没有意义的，因为拷贝出来的对象和被拷贝的对象不一样。

我们先使用父类的拷贝构造器和子类的拷贝构造器都是默认情况下进行实现：
#include
using namespace std;

class A
{
public:
    A(int x = 22)
    {
        a = x;
        cout << "A()" << endl;
    }
    int a;
};

class B : public A
{
public:
    B(int z, int y)
        :A(z)
    {
        b = y;
        cout << "B()" << endl;
    }
    int b;
};


int main()
{
    B b(666, 100);
    B bb(b);
    cout << bb.a << endl;
    cout << bb.b << endl;
    return 0;
}

运行结果为：


但是上面的方法系统默认提供的拷贝构造只能用于浅拷贝。那么如果涉及深拷贝的时候父类和子类都是默认的拷贝构造将不能实现派生类的拷贝构造。

那么接下来我们进行特殊处理来解决：
我们在这里显式的调用父类拷贝构造器。

#include <iostream>
using namespace std;


class A
{
public:
    A(int x = 22)
    {
        a = x;
        cout << "A()" << endl;
    }

    A(const A & another)
    {
        this->a = another.a;
        cout << "A(const A & another)" << endl;
    }
    int a;
};

class B: public A
{
public:
    B(int x,int y)
        :A(x)
    {
        b = y;
        cout << "B()" << endl;
    }

    B(const B & another)
        :A(another)   //在这里显示的调用父类的拷贝构造器
    {
        this->b = another.b;
        cout << "B(const B & another)" << endl;
    }
    int b;
};


int main()
{
    B b(666,100);
    B bb(b);
    cout << bb.a << endl;
    cout << bb.b << endl;
    return 0;
}

运行结果为：

通过显式的调用父类的拷贝构造器，那么我们就实现了派生类的拷贝构造。

在上面代码中我们已经实现了
当父类和子类都没有拷贝构造的时候会直接实现拷贝构造，
因为浅拷贝会默认实现。

但是如果我们使用到深拷贝的时候如何实现，就需要我们显式调用父类的拷贝构造自实现。

那么接下来我们自己实现上面的拷贝构造过程：

#include <iostream>

using namespace std;

class Student
{
public:
Student(string sn, char cs, float fs)
:name_(sn), sex_(cs), score_(fs) {}

void dis()
{
cout << "name_:" << name_ << endl;
cout << "sex_:" << sex_ << endl;
cout << "score_:" << score_ << endl;
}

Student(const Student& another)
{
this->name_ = another.name_;
this->score_ = another.score_;
this->sex_ = another.sex_;
}
private:
string name_;
char sex_;
float score_;
};

class Graduate : public Student
{
public:
Graduate(string sn,char cs,float fs,float fsa)
:Student(sn,cs,fs)
{
salary_ = fsa;
}

void dump()
{
dis();
cout << "salary:" << salary_ << endl;
}

Graduate(const Graduate& another)
:Student(another) //显示调用父类的拷贝构造器
{
this->salary_ = another.salary_;
}
private:  
float salary_;
};
int main()
{
Graduate g("zhaosi", 's', 99, 1000);
g.dump();
Graduate gg(g);
gg.dump();
return 0;
}

运行结果为：

我们前面已经说过，如果子类单独自实现拷贝构造，那就会直接调用父类的构造器而不会调用父类的拷贝构造器。

图示

#include <iostream>
using namespace std;



class C
{
public:
    C()
    {
        cout << "C()" << endl;
    }
    int a;
};

class A
{
public:
    A(int x = 22)
    {
        a = x;
        cout << "A()" << endl;
    }

    A(const A & another)
    {
        this->a = another.a;
        cout << "A(const A & another)" << endl;
    }
    int a;
};

class B: public A
{
public:
    B(int x,int y)
        :A(x)
    {
        b = y;
        cout << "B()" << endl;
    }

    B(const B & another)
        :A(another)   //在这里显示的调用父类的拷贝构造器
    {
        this->b = another.b;
        cout << "B(const B & another)" << endl;
    }
    C c;
    int b;
};


int main()
{
    B b(666,100);
    B bb(b);
    cout << bb.a << endl;
    cout << bb.b << endl;
    return 0;
}

运行结果为：

那么上面的结果我们就可以看出来：
在构造的过程中先实现父类的构造，再实现内嵌子对象的构造，在实现子类构造。
那么在拷贝构造的过程中先实现父类的拷贝构造，再实现内嵌子对象的拷贝构造，但是由于子类的拷贝构造自实现，显式调用只调用了父类的拷贝构造没有显式调用内嵌子对象的拷贝构造，所以会直接调用内嵌子对象的构造器，之后再调用子类的拷贝构造。

我们接下来演示子类显式调用内嵌子类的构造器的情况：

#include <iostream>
using namespace std;



class C
{
public:
    C(int y = 10)
    {
        c = y;
        cout << "C()" << endl;
    }
    C(const C & another)
    {
        this->c = another.c;
        cout << "C(const C & another)" << endl;
    }
    int c;
};

class A
{
public:
    A(int x)
    {
        a = x;
        cout << "A()" << endl;
    }

    A(const A & another)
    {
        this->a = another.a;
        cout << "A(const A & another)" << endl;
    }
    int a;
};

class B: public A
{
public:
    B(int x,int z,int y)
        :A(x),c(z)
    {
        b = y;
        cout << "B()" << endl;
    }

    B(const B & another)
        :A(another),c(another.c)//在这里显示的调用父类的拷贝构造器
    {
        this->b = another.b;
        cout << "B(const B & another)" << endl;
    }
    C c;
    int b;
};


int main()
{
    B b(666,100,200);
    B bb(b);
    cout << bb.a << endl;
    cout << bb.b << endl;
    cout << bb.c.c << endl;

    return 0;
}

运行结果为：

上面代码综合了派生类的构造规则和拷贝构造规则。

最后我们把派生类的构造和拷贝构造放在一起完成之前的代码：

#include <iostream>
using namespace std;
class Student
{
public:
    Student(string name, int num, char sex)
        :_name(name), _num(num), _sex(sex) {}
    void dump()
    {
        cout << "name:" << _name << endl;
        cout << "num :" << _num << endl;
        cout << "sex :" << _sex << endl;
    }
private:
    string _name;
    int _num;
    char _sex;
};
class Birthday
{
public:
    Birthday(int y, int m, int d)
        :_year(y), _month(m), _day(d)
    {
    }
    void disBirthday()
    {
        cout << "birth date:" << _year << ":" << _month << ":" << _day << endl;
    }
private:
    int _year;
    int _month;
    int _day;
};
class Graduate :public Student
{
public:
    Graduate(string name, int num, char sex, float salary, int y, int m, int d)
        :Student(name, num, sex), _salary(salary), birth(y, m, d)
    {
        _salary = salary;
    }
    Graduate(const Graduate& another)
        :Student(another), birth(another.birth)
    {
        _salary = another._salary;
    }
    void dis()
    {
        dump();
        cout << "salary:" << _salary << endl;
        birth.disBirthday();
    }
private:
    float _salary;
    Birthday birth;
};
class Doctor :public Graduate
{
public:
    Doctor(string name, int num, char sex,
        float salary, int y, int m, int d, string t)
        :Graduate(name, num, sex, salary, y, m, d), title(t)
    {
    }
private:
    string title;
};

int main()
{

    Graduate g("sun", 2001, 's', 10000, 1990, 9, 19);
    g.dis();
    Graduate gg(g);
    gg.dis();
    return 0;
}

小结

存在内嵌子对象的时候，如果子类的拷贝构造没有自实现是默认拷贝构造，那么在调用内嵌子对象类的时候默认调用内嵌类的拷贝构造器。

如果子对象自实现了拷贝构造器但是没有显示的调用内嵌子类的拷贝构造器，那么会直接调用内嵌子类的构造器。

如果子类的拷贝构造器自实现，那么如果显示的调用内嵌子类的拷贝构造器，那就直接调用内嵌子类的拷贝构造器。

父类中，一部分成员需要拷贝构造来完成，子类中，也有一部成员需要拷贝构造
来完成。子类中的内嵌子对象也需要拷贝构造来完成。
